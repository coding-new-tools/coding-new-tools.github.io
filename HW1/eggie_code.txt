import turtle
t = turtle.Turtle()
for x in xrange(2):
  for i in xrange(8):
    for j in xrange(4):
      t.pendown()
      t.forward(20)
      t.penup()
      t.forward(20)
    t.right(180)
    t.forward(160)
    t.left(90)
    t.forward(20)
    t.left(90)
    t.pendown()
  t.penup()
  t.left(90)
  t.forward(20)
